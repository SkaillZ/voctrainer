200  string[] keySplit = key.Split('.');
201  if (keySplit.Length < querySplit.Length)
202  return false;
203  string currentSegment = keySplit[querySplit.Length - 1];
204  return key.StartsWith(query, StringComparison.CurrentCultureIgnoreCase) && currentSegment != query;
205  })
206  .Select(key =>
